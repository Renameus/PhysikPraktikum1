a&=0,0050\frac{1}{\text{s V}} \label{eqalinrega}\\
\Delta a&=38,0\% \\
b&=101,4606\frac{1}{\text{s}}\\
\Delta b&=0,9\%
\end{align}

b&=-3,1548 * 10^10\\

\end{align}
